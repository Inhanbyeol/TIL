```js
let b = 1;

function hi () {

  const a = 1;
  let b = 100;
  
  b++;

  console.log(a,b); // 1, 101 hi 함수 외부 최상단에 선언된 'b' 변수를 가져올 수 있으나, 함수 내 동일한 명칭으로 선언 된 변수가 있기에, b(100)값을, 그리고 b++ 증감 연사자를 통해 b(101)을 출력한다.

}

// console.log(a); 지역변수인 a를 지역을 벗어난 곳에서 호출 시 값을 가져올 수 없으며 오류를 노출한다 'a is not defined'

console.log(b); //1 최상단에 선언한 b 변수의 값을 가져오며, hi() 함수 내부의 지역변수는 불러올 수 없다.

hi(); //1, 101

console.log(b); //1
```

```js
1 == "1"; //true '==' 방식은 값에 대한 결과만 비교하는 방식이며, 값의 타입은 비교하지 않는다.
1 === "1"; //false '===' 방식은 값 뿐만 아니라 값의 타입 또한 동일할 경우로써, 정수와 문자열이기에 false 결과를 보여준다.
```
